TO TRANSFORM REACT APP TO PWA

a. npm install -D vite-plugin-pwa   //run from frontend

b. //configure vite.configure.js, copypaste the code below 

import { defineConfig } from "vite";
import react from "@vitejs/plugin-react";
import mkcert from "vite-plugin-mkcert";
import { VitePWA } from "vite-plugin-pwa";

export default defineConfig({
  plugins: [
    react(),
    mkcert(),
    VitePWA({
      registerType: "autoUpdate",
      devOptions: {
        enabled: true, // 👈 enable PWA in dev mode
      },
      manifest: {
        name: "My React PWA",
        short_name: "ReactPWA",
        description: "My Vite React Progressive Web App",
        theme_color: "#ffffff",
        background_color: "#ffffff",
        display: "standalone",
        start_url: "/",
        icons: [
          {
            src: "/pwa-192x192.png",
            sizes: "192x192",
            type: "image/png",
          },
          {
            src: "/pwa-512x512.png",
            sizes: "512x512",
            type: "image/png",
          },
          {
            src: "/pwa-512x512-maskable.png",
            sizes: "512x512",
            type: "image/png",
            purpose: "maskable",
          },
        ],
      },
    }),
  ],
  server: {
    https: true,
  },
});



c.Add icons: one 192 and two 512 in dimension, png in format in public folder



d.Copy and paste the ff to main.jsx


import { registerSW } from "virtual:pwa-register";

registerSW({
  onNeedRefresh() {
    console.log("New content available");
  },
  onOfflineReady() {
    console.log("App ready to work offline");
  },
});




e.npm run dev

f.TO AVOID CONFLIFT WITH OTHER PWA that are already installed;
   In vite.config.js manifest, configure the following;
id:"pwa-v2",  /add an id
start_url: "/"    to start_url: "/pwa/"
add scope: "/pwa/"
