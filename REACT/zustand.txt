**WITH PERSIST, MEANING SAVE EITHER IN SESSION OR LOCALSTORAGE

import { create } from "zustand";
import { persist } from "zustand/middleware";

export const useMyData = create(
  persist(
    (set) => ({
      mydata: [],
      setMyData: (data) => set({ mydata: data }),
      clearMyData: () => set({ mydata: null }),
    }),
    {
      name: "mydata-storage", // key in storage
    },
  ),
);
